from langchain_core.tools import tool, ToolException
from pathlib import Path

# ── Whitelist stricte des commandes autorisées ──────────────────
COMMANDES_AUTORISEES = {
# Navigation
“ls”, “pwd”, “find”, “tree”,
# Lecture
“cat”, “head”, “tail”, “grep”, “wc”, “diff”,
# Système info
“df”, “du”, “ps”, “top”, “free”, “uname”, “uptime”,
# Réseau info (pas d’envoi)
“ping”, “nslookup”, “host”,
# Dev
“python3”, “pip”, “node”, “npm”,
“git”, “docker”, “make”,
# Manipulation fichiers (pas rm!)
“mkdir”, “cp”, “mv”, “touch”, “echo”,
}

# Patterns INTERDITS — bloqués même si la commande est whitelistée
PATTERNS_DANGEREUX = [
r”rm\s+-rf”,           # rm -rf
r”>\s*/dev/”,           # redirection vers /dev
r”curl.+|\s*bash”,     # curl | bash
r”wget.+|\s*sh”,       # wget | sh
r”sudo”,                # élévation de privilèges
r”chmod\s+[0-7]*7”,     # chmod world-writable
r”;\s*rm”,              # commande chaînée avec rm
r”&&\s*rm”,             # commande chaînée avec rm
r”eval\s”,              # eval
r”/etc/passwd”,         # fichiers système sensibles
r”/etc/shadow”,
]

WORKSPACE = Path(”./agent_workspace”).resolve()

def valider_commande(cmd: str) -> tuple[bool, str]:
””“Valide une commande — retourne (ok, message).”””
# 1. Vérifier les patterns dangereux
for pattern in PATTERNS_DANGEREUX:
if re.search(pattern, cmd, re.IGNORECASE):
return False, f“Pattern interdit détecté: {pattern}”

```
# 2. Extraire la commande de base
try:
    tokens  = shlex.split(cmd)
    base    = os.path.basename(tokens[0]) if tokens else ""
except ValueError as e:
    return False, f"Commande invalide: {e}"

# 3. Vérifier la whitelist
if base not in COMMANDES_AUTORISEES:
    return False, f"Commande non autorisée: '{base}'. Autorisées: {sorted(COMMANDES_AUTORISEES)}"

return True, "OK"
```

2.3 — Human-in-the-Loop avant les actions dangereuses

02_hitl.py
python
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
from langgraph.prebuilt import ToolNode, tools_condition
from langchain_ollama import ChatOllama
from langchain_core.messages import HumanMessage, AIMessage
from typing import TypedDict, Annotated
import operator

class ShellState(TypedDict):
messages: Annotated[list, operator.add]

# Tools qui nécessitent validation humaine
tools_dangeureux = [git_add_commit, git_create_branch]
tools_sûrs       = [git_status, git_diff, git_log, executer_commande]
all_tools        = tools_sûrs + tools_dangeureux

llm   = ChatOllama(model=“llama3.2”).bind_tools(all_tools)
mem   = MemorySaver()

def agent_node(state: ShellState):
return {“messages”: [llm.invoke(state[“messages”])]}

g = StateGraph(ShellState)
g.add_node(“agent”, agent_node)
g.add_node(“tools”, ToolNode(all_tools))
g.set_entry_point(“agent”)
g.add_conditional_edges(“agent”, tools_condition)
g.add_edge(“tools”, “agent”)

# ── interrupt_before=“tools” → pause pour validation humaine ────
shell_agent = g.compile(
checkpointer=mem,
interrupt_before=[“tools”]  # PAUSE avant chaque tool call
)

def run_with_approval(question: str, thread_id: str = “shell-1”):
cfg = {“configurable”: {“thread_id”: thread_id}}

```
# Phase 1: L'agent prépare ses actions
shell_agent.invoke({"messages": [HumanMessage(question)]}, cfg)

# Inspecter ce que l'agent veut faire
state   = shell_agent.get_state(cfg)
last_msg = state.values["messages"][-1]

if hasattr(last_msg, "tool_calls") and last_msg.tool_calls:
    print("\n🔔 L'agent veut exécuter:")
    for tc in last_msg.tool_calls:
        print(f"  [{tc['name']}] {tc['args']}")
    
    approval = input("\nValider? (o/N): ").strip().lower()
    
    if approval == "o":
        # Phase 2: Reprendre l'exécution
        result = shell_agent.invoke(None, cfg)  # None = reprendre
        return result["messages"][-1].content
    else:
        return "Action annulée par l'utilisateur."

return state.values["messages"][-1].content
```

Quiz — Module 2 : Terminal

2 questions
Q1 · QCM
Pourquoi une whitelist est-elle préférable à une blacklist pour sécuriser l'accès terminal ?
ALa whitelist est plus facile à maintenir
BLa blacklist est impossible à compléter — il y a toujours une commande dangereuse oubliée
CLa whitelist permet plus de commandes
DLes deux sont équivalentes
✓ B. Une blacklist suppose de connaître toutes les commandes dangereuses — impossible. Un attaquant peut toujours trouver une commande non bloquée (python3 -c "import os; os.system('rm -rf /')"). La whitelist inverse le problème : rien n'est autorisé sauf ce qui est explicitement listé.
Q2 · Ouverte
Expliquez le mécanisme interrupt_before de LangGraph et pourquoi il est essentiel pour les agents qui agissent sur des systèmes réels.
✓ interrupt_before=["tools"] pause l'exécution du graphe juste avant l'appel d'un outil — grâce au checkpointer qui sauvegarde l'état. L'humain peut inspecter ce que l'agent veut faire, valider ou refuser. En appelant invoke(None, cfg) on reprend exactement là où on s'était arrêté.

C'est essentiel car les agents LLM font des erreurs. Sur un système réel (git, DB, filesystem), une action irréversible mal comprise peut causer des dommages permanents. Human-in-the-Loop est la première ligne de défense.

def analyser_ast(code: str) -> tuple[bool, str]:
””“Analyse statique du code pour détecter les patterns dangereux.”””
try:
tree = ast.parse(code)
except SyntaxError as e:
return False, f“Erreur de syntaxe: {e}”

```
for node in ast.walk(tree):
    # Interdit: import os, import sys, import subprocess...
    if isinstance(node, (ast.Import, ast.ImportFrom)):

            mod = alias.name.split(".")[0]
            if mod not in SAFE_MODULES:
                return False, f"Import interdit: {alias.name}"
    # Interdit: exec(), eval(), compile()
    if isinstance(node, ast.Call):
        if isinstance(node.func, ast.Name):

                return False, f"Appel interdit: {node.func.id}()"
```

Réflexion & Auto-Critique

Le pattern Reflexion permet à l'agent de critiquer sa propre réponse et de s'améliorer itérativement avant de livrer.

Expert

5.1 — Le pattern Reflexion (Shinn et al. 2023)

🪞

Idée centrale

Un agent qui génère une réponse, puis joue le rôle d'un critique pour l'améliorer. Le critique donne un score et des suggestions — le générateur réécrit. Après N itérations, la meilleure réponse est retournée. Gain typique : +40% sur les tâches d'écriture et de code.

05_reflexion.py
python
from langgraph.graph import StateGraph, END
from pydantic import BaseModel, Field
from typing import TypedDict, Annotated
import operator

llm = ChatOllama(model=“llama3.2”, temperature=0.4)

# ── Schéma de critique ───────────────────────────────────────────
class Critique(BaseModel):
score:          int  = Field(ge=1, le=10, description=“Score de qualité (1-10)”)
points_forts:   list[str] = Field(description=“Ce qui est bien”)
ameliorations:  list[str] = Field(description=“Ce qui peut être amélioré”)
pret_a_livrer: bool = Field(description=“True si score >= 8 et rien d’essentiel à corriger”)

class ReflexionState(TypedDict):
tache:      str
reponse:    str
critiques:  Annotated[list[str], operator.add]
iterations: int

# ── LLM critique ─────────────────────────────────────────────────
critique_chain = (
ChatPromptTemplate.from_messages([
(“system”, ””“Tu es un critique expert et exigeant.
Évalue la réponse avec précision.
Score 8+ seulement si vraiment excellent.
Sois constructif et précis dans tes suggestions.”””),
(“human”, “Tâche: {tache}\n\nRéponse à évaluer:\n{reponse}\n\nCritique:”),
])
| llm.with_structured_output(Critique)
)

# ── LLM améliorateur ─────────────────────────────────────────────
improve_chain = (
ChatPromptTemplate.from_messages([
(“system”, “Tu améliores les réponses en tenant compte des critiques reçues.”),
(“human”, ””“Tâche: {tache}

Réponse précédente:
{reponse}

Critique reçue:
{critique}

Réponse améliorée:”””),
])
| llm | StrOutputParser()
)

# ── Nœuds du graphe ──────────────────────────────────────────────
def generer(state: ReflexionState) -> dict:
””“Génère ou améliore la réponse.”””
if not state.get(“reponse”):
# Première génération
reponse = llm.invoke(state[“tache”]).content
else:
# Amélioration basée sur la dernière critique
derniere_critique = state[“critiques”][-1] if state[“critiques”] else ””
reponse = improve_chain.invoke({
“tache”:    state[“tache”],
“reponse”:  state[“reponse”],
“critique”: derniere_critique
})
return {“reponse”: reponse, “iterations”: state.get(“iterations”, 0) + 1}

def critiquer(state: ReflexionState) -> dict:
””“Critique la réponse courante.”””
c = critique_chain.invoke({“tache”: state[“tache”], “reponse”: state[“reponse”]})
critique_txt = (
f“Score: {c.score}/10\n”
f“Points forts: {’, ‘.join(c.points_forts)}\n”
f“À améliorer: {’, ’.join(c.ameliorations)}”
)
print(f“🔍 Critique itération {state.get(‘iterations’,0)}: score={c.score}/10”)
return {“critiques”: [critique_txt], “pret_a_livrer”: c.pret_a_livrer}

def router(state: ReflexionState) -> str:
if state.get(“pret_a_livrer”) or state.get(“iterations”, 0) >= 3:
return “fin”
return “ameliorer”

g = StateGraph(ReflexionState)
g.add_node(“generer”,  generer)
g.add_node(“critiquer”, critiquer)
g.set_entry_point(“generer”)
g.add_edge(“generer”, “critiquer”)
g.add_conditional_edges(“critiquer”, router, {“ameliorer”:“generer”, “fin”:END})

reflexion = g.compile()
result = reflexion.invoke({
“tache”: “Écris une fonction Python de tri rapide (quicksort) avec explications”,
“reponse”: ””, “critiques”: [], “iterations”: 0
})
print(f“✅ Réponse finale après {result[‘iterations’]} itérations:”)
print(result[“reponse”])

from langgraph.graph import StateGraph, END
from langgraph.prebuilt import ToolNode, tools_condition
from langchain_core.tools import tool
from langchain_core.messages import AIMessage, HumanMessage
from typing import TypedDict, Annotated
import operator

# ── 1. Handoff tools ─────────────────────────────────────────────
# Chaque agent a un tool pour transférer vers un autre agent
@tool
def vers_qa() -> str:
””“Transfère la tâche à l’agent QA pour les tests et la revue qualité.
Utilise quand le code est écrit et doit être testé.”””
return “Transfert vers QA initié”

@tool
def vers_developpeur() -> str:
””“Transfère la tâche au développeur pour écrire ou corriger du code.
Utilise quand des bugs sont trouvés ou que du code doit être créé.”””
return “Transfert vers Développeur initié”

@tool
def vers_devops() -> str:
””“Transfère la tâche à DevOps pour le déploiement ou git.
Utilise quand le code est validé et prêt à être déployé.”””
return “Transfert vers DevOps initié”

# ── 2. État et routing basé sur tool_calls ───────────────────────
class SwarmState(TypedDict):
messages:      Annotated[list, operator.add]
agent_courant: str

def make_swarm_agent(llm_instance, tools_list: list, system: str):
””“Crée un nœud agent swarm.”””
bound_llm = llm_instance.bind_tools(tools_list)
def node(state: SwarmState) -> dict:
msgs = [(“system”, system)] + state[“messages”]
return {“messages”: [bound_llm.invoke(msgs)]}
return node

node_dev = make_swarm_agent(
ChatOllama(model=“llama3.2”, temperature=0.2),
[executer_python, ecrire_fichier, vers_qa, vers_devops],
“Tu es développeur Python. Crée du code, puis transfère vers QA pour les tests.”
)

ChatOllama(model=“llama3.2”, temperature=0),
[executer_python, lire_fichier, vers_developpeur, vers_devops],
“Tu es QA. Teste le code. Si bugs → vers_developpeur. Si OK → vers_devops.”
)
node_devops = make_swarm_agent(
ChatOllama(model=“llama3.2”, temperature=0),
[executer_commande, git_add_commit, vers_developpeur],
“Tu es DevOps. Commite et déploie. Si problème → vers_developpeur.”
)

# ── 3. Router basé sur le dernier tool_call ──────────────────────
def swarm_router(state: SwarmState) -> str:
””“Route vers l’agent demandé par le handoff tool.”””
last = state[“messages”][-1]
if not hasattr(last, “tool_calls”) or not last.tool_calls:
return “end”
tool_name = last.tool_calls[0][“name”]
return {
“vers_qa”:          “qa”,
“vers_developpeur”: “dev”,
“vers_devops”:      “devops”,
}.get(tool_name, “end”)

def tool_executor(state: SwarmState) -> dict:
””“Exécute le tool courant (métier ou handoff).”””
all_tools_map = {
“executer_python”:   executer_python,
“ecrire_fichier”:    ecrire_fichier,
“lire_fichier”:      lire_fichier,
“git_add_commit”:    git_add_commit,
“vers_qa”:           vers_qa,
“vers_developpeur”:  vers_developpeur,
“vers_devops”:       vers_devops,
}
last = state[“messages”][-1]
results = []
for tc in last.tool_calls:
fn = all_tools_map.get(tc[“name”])
if fn:
r = fn.invoke(tc[“args”])
from langchain_core.messages import ToolMessage
results.append(ToolMessage(content=str(r), tool_call_id=tc[“id”]))
return {“messages”: results}

# ── 4. Graphe swarm ──────────────────────────────────────────────
g = StateGraph(SwarmState)
g.add_node(“dev”,    node_dev)
g.add_node(“qa”,     node_qa)
g.add_node(“devops”, node_devops)
g.add_node(“tools”,  tool_executor)

g.set_entry_point(“dev”)

# Chaque agent → tools → router → prochain agent ou END
for node in [“dev”, “qa”, “devops”]:
g.add_edge(node, “tools”)

g.add_conditional_edges(“tools”, swarm_router,
{“dev”:“dev”, “qa”:“qa”, “devops”:“devops”, “end”:END})

swarm_graph = g.compile()
result = swarm_graph.invoke({
“messages”: [HumanMessage(“Crée un script de tri et déploie-le”)],
“agent_courant”: “dev”
})
