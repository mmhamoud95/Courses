python –version   # Doit être >= 3.8

# Installer Streamlit

# Vérifier l’installation
streamlit –version

streamlit hello

```
Utilisez toujours un environnement virtuel : python -m venv env puis source env/bin/activate (Linux/Mac) ou env\Scripts\activate (Windows).

Créez un fichier app.py et copiez ce code minimal :

```

```
⚠️

Le comportement de réexécution

Chaque interaction avec un widget déclenche une réexécution complète du script. Les variables sont recalculées. Pour conserver l'état entre les exécutions, utilisez st.session_state (module 7).
```

```
🧠 session_state — Mémoire entre exécutions

PYTHON — compteur.py
import streamlit as st
```

# Initialiser la variable d’état
if “compteur” not in st.session_state:
st.session_state.compteur = 0

st.title(“🔢 Compteur persistant”)
st.metric(“Valeur”, st.session_state.compteur)

if col1.button(“➕ +1”):
st.session_state.compteur += 1
if col2.button(“➖ -1”):

```
⚡ @st.cache_data — Mise en cache des données

PYTHON
import streamlit as st
```

import pandas as pd
import time

# Sans cache : rechargé à chaque interaction (lent)
def charger_donnees_lent():
time.sleep(3)  # Simule une requête SQL lente
return pd.read_csv(“data.csv”)

# Avec cache : exécuté UNE SEULE FOIS, puis mis en cache
@st.cache_data
def charger_donnees_rapide():
time.sleep(3)  # Seulement au 1er appel
return pd.read_csv(“data.csv”)

# Cache avec TTL (expiration automatique)
@st.cache_data(ttl=3600)  # Cache valide 1 heure
def get_api_data(url):
return requests.get(url).json()

# @st.cache_resource pour les connexions DB, modèles ML
@st.cache_resource
def charger_modele_ml():
import joblib
return joblib.load(“model.pkl”)
```
🧠

cache_data vs cache_resource

@st.cache_data : pour les données (DataFrames, JSON, listes). Chaque utilisateur reçoit une copie.
@st.cache_resource : pour les ressources partagées (connexions DB, modèles ML). Un seul objet partagé entre tous les utilisateurs.
```

```
Convention de nommage

Les fichiers dans pages/ sont ordonnés par leur préfixe numérique. Les emojis dans le nom apparaissent dans la navigation. Ex : 1_📊_Dashboard.py → "📊 Dashboard" dans le menu.

import streamlit as st
```

# Configuration globale de l’app
st.set_page_config(
page_title=“Mon Dashboard”,
page_icon=“📊”,
layout=“wide”,
initial_sidebar_state=“expanded”
)

st.title(“🏠 Accueil”)

# Partager des données entre pages via session_state
if “utilisateur” not in st.session_state:
st.session_state.utilisateur = “Visiteur”

nom = st.text_input(“Votre nom”, st.session_state.utilisateur)
st.session_state.utilisateur = nom

import streamlit as st

@st.cache_data
def charger_donnees(fichier_bytes, nom_fichier):
””“Charge et prépare le CSV. Mis en cache automatiquement.”””
df = pd.read_csv(fichier_bytes)
df[“date”] = pd.to_datetime(df[“date”])
df[“ca”] = df[“quantite”] * df[“prix_unitaire”]
return df

def calculer_kpis(df):
””“Calcule les KPIs principaux.”””
return {
“ca_total”: df[“ca”].sum(),
“nb_commandes”: len(df),
“panier_moyen”: df[“ca”].mean(),
“nb_produits”: df[“produit”].nunique()
}

import plotly.express as px
from utils.data_utils import calculer_kpis

st.title(“📊 Dashboard Principal”)

# Vérifier que les données sont chargées
if “df” not in st.session_state:
st.warning(“⚠️ Importez d’abord un fichier sur la page Accueil.”)
st.stop()

df = st.session_state[“df”]

# === SIDEBAR : FILTRES ===
with st.sidebar:
st.header(“🔍 Filtres”)

```
date_range = st.date_input(
    "Période",
    value=(df["date"].min(), df["date"].max())
)

categories = st.multiselect(
    "Catégories",
    options=df["categorie"].unique(),
    default=df["categorie"].unique()
)
```

# === FILTRAGE ===
mask = (
(df[“date”].dt.date >= date_range[0]) &
(df[“date”].dt.date <= date_range[1]) &
(df[“categorie”].isin(categories))
)
df_filtre = df[mask]

# === KPIs ===
kpis = calculer_kpis(df_filtre)
col1, col2, col3, col4 = st.columns(4)
col1.metric(“💰 CA Total”, f”€ {kpis[‘ca_total’]:,.0f}”)
col2.metric(“📦 Commandes”, f”{kpis[‘nb_commandes’]:,}”)
col3.metric(“🛒 Panier Moyen”, f”€ {kpis[‘panier_moyen’]:.2f}”)
col4.metric(“🏷️ Produits”, kpis[‘nb_produits’])

st.divider()

# === GRAPHIQUES ===
col_left, col_right = st.columns([2, 1])

with col_left:
# Évolution du CA par mois
df_mensuel = df_filtre.groupby(df_filtre[“date”].dt.to_period(“M”).astype(str))[“ca”].sum().reset_index()
fig_line = px.line(
df_mensuel, x=“date”, y=“ca”,
title=“📈 Évolution mensuelle du CA”,
labels={“ca”: “CA (€)”, “date”: “Mois”},
template=“plotly_dark”
)
st.plotly_chart(fig_line, use_container_width=True)

with col_right:
# Répartition par catégorie
df_cat = df_filtre.groupby(“categorie”)[“ca”].sum().reset_index()
fig_pie = px.pie(
df_cat, values=“ca”, names=“categorie”,
title=“🥧 CA par catégorie”,
template=“plotly_dark”
)
st.plotly_chart(fig_pie, use_container_width=True)
