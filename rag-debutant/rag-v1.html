✅ Apprendre (20 min)

Le minimum vital

  • Un embedding = une position dans l'espace. "Chat" et "félin" sont proches. "Voiture" est loin.
  • Similarité cosinus = l'angle entre deux vecteurs. Score de 0 (sans rapport) à 1 (identique).
  • Pourquoi ça bat la recherche par mots-clés : "procédure incident" trouve "gestion des pannes" même sans mot commun.
  • Modèle à utiliser : all-MiniLM-L6-v2 — gratuit, local, 90% aussi bon qu'OpenAI pour débuter.
❌ Ignorer (80% du temps)

Superflu pour J1

  • Les 384 dimensions du vecteur et leur signification
  • Différences entre modèles (BERT, MPNet, E5…)
  • Fine-tuning d'un modèle d'embedding
  • Visualisation t-SNE / PCA des vecteurs
  • Dot product vs cosine similarity — théorie

```

from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

model = SentenceTransformer(‘all-MiniLM-L6-v2’)  # ~90MB, téléchargé une fois

# ── FIL ROUGE : tes vrais documents d’entreprise ──────────────────────
# Remplace par des extraits réels de tes docs (max 2-3 phrases / entrée)
DOCS_ENTREPRISE = [
“Politique de sécurité : tout accès VPN requiert une authentification MFA.”,
“Guide onboarding IT : créer le compte AD, affecter les licences Office365.”,
“Procédure incident P1 : alerter le NOC dans les 15 min, ouvrir un ticket P1.”,
“Charte télétravail RH : 2 jours max par semaine, accord manager obligatoire.”,
“RGPD : toute collecte de données personnelles nécessite une base légale.”,
“Backup : snapshots quotidiens à 2h du matin, rétention 30 jours.”,
]

# ── ÉTAPE 1 : Transformer le texte en vecteurs ─────────────────────────
vectors = model.encode(DOCS_ENTREPRISE)  # shape: (6, 384) — 6 docs × 384 dimensions
print(f”✓ {len(DOCS_ENTREPRISE)} documents → vecteurs {vectors.shape}”)

# ── ÉTAPE 2 : Recherche sémantique sur une requête ─────────────────────
question = “Comment gérer une panne critique en urgence ?”
q_vector = model.encode([question])  # même opération pour la requête

scores = cosine_similarity(q_vector, vectors)[0]  # score entre 0 et 1 pour chaque doc

# ── RÉSULTATS : triés par pertinence ───────────────────────────────────
ranked = sorted(enumerate(scores), key=lambda x: x[1], reverse=True)

print(f”\n🔍 Question : ‘{question}’\n”)
for rank, (idx, score) in enumerate(ranked[:3], 1):
print(f”  #{rank} ({score:.2f}) → {DOCS_ENTREPRISE[idx][:70]}…”)

# ── CE QUE TU DOIS OBSERVER ────────────────────────────────────────────
# → “Procédure incident P1” devrait être en #1 (score > 0.5)
# → “Backup” peut apparaître en #2 (notion d’“urgence” liée)
# → “Charte télétravail” devrait être en bas (hors sujet)


# SORTIE ATTENDUE :

# #1 (0.67) → Procédure incident P1 : alerter le NOC dans les 15 min…

# #2 (0.41) → Backup : snapshots quotidiens à 2h du matin…

# #3 (0.28) → Guide onboarding IT : créer le compte AD…
```

import chromadb
from sentence_transformers import SentenceTransformer

# ── MÊME CORPUS QUE J1 — on le stocke maintenant ──────────────────────
DOCS = [
{“id”: “sec_01”,  “text”: “Politique sécurité : tout accès VPN requiert MFA.”,            “dept”: “IT”,  “source”: “politique_securite_v3.pdf”},
{“id”: “it_01”,   “text”: “Onboarding IT : créer compte AD, affecter licences Office365.”, “dept”: “IT”,  “source”: “guide_onboarding.pdf”},
{“id”: “inc_01”,  “text”: “Incident P1 : alerter le NOC dans les 15 min.”,               “dept”: “IT”,  “source”: “proc_incident.pdf”},
{“id”: “rh_01”,   “text”: “Télétravail : 2 jours max/semaine, accord manager requis.”,    “dept”: “RH”,  “source”: “charte_teletravail.pdf”},
{“id”: “rgpd_01”, “text”: “RGPD : toute collecte de données requiert une base légale.”,   “dept”: “JUR”, “source”: “note_rgpd_2024.pdf”},
{“id”: “bkp_01”,  “text”: “Backup : snapshots quotidiens 2h, rétention 30 jours.”,        “dept”: “IT”,  “source”: “politique_backup.pdf”},
]

# ── INIT : base persistante sur disque ────────────────────────────────
client = chromadb.PersistentClient(path=”./base_entreprise”)  # dossier créé auto
col = client.get_or_create_collection(“docs_internes”)
model = SentenceTransformer(‘all-MiniLM-L6-v2’)

# ── INGESTION : éviter les doublons ──────────────────────────────────
existing_ids = col.get()[“ids”]
nouveaux = [d for d in DOCS if d[“id”] not in existing_ids]

if nouveaux:
col.add(
documents=[d[“text”]     for d in nouveaux],
embeddings=model.encode([d[“text”] for d in nouveaux]).tolist(),
metadatas=[{“dept”: d[“dept”], “source”: d[“source”]} for d in nouveaux],
ids=[d[“id”]           for d in nouveaux],
)
print(f”✓ {len(nouveaux)} docs ajoutés | Total : {col.count()}”)
else:
print(f”✓ Base déjà à jour ({col.count()} docs)”)

# ── REQUÊTE SIMPLE ───────────────────────────────────────────────────
question = “Quelles sont les règles pour travailler depuis chez soi ?”
results = col.query(
query_embeddings=model.encode([question]).tolist(),
n_results=2,
include=[“documents”, “metadatas”, “distances”]
)

for i, (doc, meta, dist) in enumerate(zip(
results[“documents”][0],
results[“metadatas”][0],
results[“distances”][0]
)):
print(f”\n#{i+1} Source: {meta[‘source’]} | Distance: {dist:.3f}”)
print(f”     {doc}”)

# ── REQUÊTE FILTRÉE PAR DÉPARTEMENT ──────────────────────────────────
print(”\n── Recherche dans IT uniquement ──”)
results_it = col.query(
query_embeddings=model.encode([“comment configurer un nouveau poste ?”]).tolist(),
n_results=2,
where={“dept”: “IT”}   # ← filtre : uniquement les docs IT
)
print(results_it[“documents”][0])
```

# Prérequis : ollama pull mistral  (une seule fois dans le terminal)

# ⚠️ LangChain v1 : RetrievalQA est supprimé → on utilise le pattern LCEL

from langchain_community.document_loaders import PyPDFLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter  # ← nouveau package v1
from langchain_chroma import Chroma                                   # ← langchain-chroma (plus community)
from langchain_huggingface import HuggingFaceEmbeddings               # ← langchain-huggingface
from langchain_ollama import ChatOllama
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

# ── ÉTAPE 1 : Charger UN vrai PDF ─────────────────────────────────────
pages = PyPDFLoader(“politique_securite_v3.pdf”).load()
print(f”✓ {len(pages)} pages chargées”)

# ── ÉTAPE 2 : Découper en chunks ──────────────────────────────────────
chunks = RecursiveCharacterTextSplitter(
chunk_size=500, chunk_overlap=50
).split_documents(pages)
print(f”✓ {len(chunks)} chunks générés”)

# ── ÉTAPE 3 : Indexer dans ChromaDB ──────────────────────────────────
vectordb = Chroma.from_documents(
documents=chunks,
embedding=HuggingFaceEmbeddings(model_name=“all-MiniLM-L6-v2”),
persist_directory=”./base_entreprise”
)
retriever = vectordb.as_retriever(search_kwargs={“k”: 3})
print(f”✓ {vectordb._collection.count()} chunks indexés”)

# ── ÉTAPE 4 : Pipeline LCEL (pattern LangChain v1) ───────────────────
# Plus de RetrievalQA → on compose avec le pipe operator |
llm = ChatOllama(model=“mistral”, temperature=0)

prompt = ChatPromptTemplate.from_messages([
(“system”, “Réponds UNIQUEMENT avec le contexte. Si absent, dis-le clairement.\n\nContexte:\n{context}”),
(“human”, ”{question}”),
])

# Fonction helper pour formater les docs récupérés
def format_docs(docs):
return ”\n\n”.join(
f”[Source: {d.metadata.get(‘source’,’?’)}]\n{d.page_content}”
for d in docs
)

# Le pipe | est le pattern LCEL v1 — lisible et composable
rag_chain = (
{“context”: retriever | format_docs, “question”: RunnablePassthrough()}
| prompt
| llm
| StrOutputParser()
)

# ── TESTER avec 3 questions types ────────────────────────────────────
questions = [
“Quelles sont les exigences d’authentification pour le VPN ?”,
“Que faire en cas de perte d’un appareil professionnel ?”,
“Quelles données sont interdites de stocker sur un poste local ?”,
]

for q in questions:
print(f”\n❓ {q}”)
# Récupérer les sources séparément pour les afficher
sources = [d.metadata.get(‘source’,’?’) for d in retriever.invoke(q)]
answer = rag_chain.invoke(q)
print(f”💬 {answer}”)
print(f”📄 Sources : {sources}”)
```

✅ La règle des 3 tailles (10 min)

  • 200 tokens → précis mais sans contexte. Bon pour FAQ courtes.
  • 500 tokens → le sweet spot pour 80% des cas. Commence toujours là.
  • 1000 tokens → bon contexte mais dilue la pertinence. Docs narratifs.
  • Overlap = 10% de chunk_size. Si chunk=500, overlap=50. Pas plus.
❌ Ignorer maintenant

  • SemanticChunker (fragile, lent, coûteux)
  • ParentDocument Retriever (complexité prématurée)
  • MarkdownHeaderTextSplitter (cas spécifique)
  • Optimisation token-level avec tiktoken

from langchain_community.document_loaders import PyPDFLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter  # ← v1
from langchain_huggingface import HuggingFaceEmbeddings                # ← v1
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

pages = PyPDFLoader(“politique_securite_v3.pdf”).load()
emb = HuggingFaceEmbeddings(model_name=“all-MiniLM-L6-v2”)

# ── Les 3 configurations à tester ─────────────────────────────────────
CONFIGS = [
{“name”: “Petit  (200/20)”,  “size”: 200,  “overlap”: 20},
{“name”: “Moyen  (500/50)”,  “size”: 500,  “overlap”: 50},  # ← sweet spot
{“name”: “Grand  (1000/100)”, “size”: 1000, “overlap”: 100},
]

# ── Question de test (utilise une vraie question sur ton doc) ──────────
QUESTION = “Que faire si un employé perd son ordinateur portable ?”
q_vec = np.array(emb.embed_query(QUESTION))

print(f”\n{‘Config’:<20} {‘Chunks’:>6} {‘Score #1’:>9} {‘Contexte chunk #1 (100 chars)’}”)
print(”-” * 80)

for cfg in CONFIGS:
splitter = RecursiveCharacterTextSplitter(chunk_size=cfg[“size”], chunk_overlap=cfg[“overlap”])
chunks = splitter.split_documents(pages)
texts = [c.page_content for c in chunks]
vecs = np.array(emb.embed_documents(texts))

```
scores = cosine_similarity([q_vec], vecs)[0]
best_idx = int(np.argmax(scores))
best_score = scores[best_idx]
best_text = texts[best_idx][:100].replace("\n", " ")

print(f"{cfg['name']:<20} {len(chunks):>6} {best_score:>9.3f} {best_text}")

from langchain_ollama import ChatOllama
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.messages import HumanMessage, AIMessage
from langchain.chains import create_history_aware_retriever, create_retrieval_chain
from langchain.chains.combine_documents import create_stuff_documents_chain
# retriever, vectordb = réutilisés depuis J3

llm = ChatOllama(model=“mistral”, temperature=0)

# ── ÉTAPE 1 : Retriever “conscient” de l’historique ───────────────────
# Reformule la question en tenant compte du contexte précédent
contextualize_prompt = ChatPromptTemplate.from_messages([
(“system”, “Reformule la question en tenant compte de l’historique. “
“Retourne UNIQUEMENT la question reformulée, rien d’autre.”),
MessagesPlaceholder(“chat_history”),   # ← placeholder pour la liste de messages
(“human”, ”{input}”),
])
history_aware_retriever = create_history_aware_retriever(llm, retriever, contextualize_prompt)

# ── ÉTAPE 2 : Chaîne RAG avec prompt anti-hallucination ───────────────
qa_prompt = ChatPromptTemplate.from_messages([
(“system”, “Réponds UNIQUEMENT avec le contexte. Si absent : “
“‘Information non trouvée dans les documents.’\n\nContexte:\n{context}”),
MessagesPlaceholder(“chat_history”),
(“human”, ”{input}”),
])
question_answer_chain = create_stuff_documents_chain(llm, qa_prompt)
rag_conv = create_retrieval_chain(history_aware_retriever, question_answer_chain)

# ── ÉTAPE 3 : L’historique = une simple liste — tu le gères toi-même ──
# Plus de ConversationBufferWindowMemory → plus simple, plus transparent
chat_history = []   # liste de HumanMessage / AIMessage
MAX_HISTORY  = 5    # garde les 5 derniers tours (pareto : k=5 suffit)

conversation = [
“Quelles sont les règles d’accès VPN de l’entreprise ?”,
“Est-ce que MFA s’applique aussi en dehors du VPN ?”,   # référence implicite
“Et si quelqu’un ne respecte pas ces règles ?”,            # suite logique
]

for question in conversation:
result = rag_conv.invoke({“input”: question, “chat_history”: chat_history})
answer = result[“answer”]
print(f”\n👤 {question}”)
print(f”🤖 {answer}”)
# Mettre à jour l’historique + tronquer à MAX_HISTORY tours
chat_history.extend([HumanMessage(content=question), AIMessage(content=answer)])
chat_history = chat_history[-MAX_HISTORY * 2:]   # *2 car 1 tour = 2 messages
print(f”── historique : {len(chat_history)} messages”)

# Prérequis : ollama serve  (terminal séparé)  |  ollama pull mistral

# Lancer : streamlit run jour7_app.py

# ⚠️ LangChain v1 : ConversationalRetrievalChain/Memory supprimés → create_retrieval_chain + liste messages

import streamlit as st, tempfile
from langchain_community.document_loaders import PyPDFLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter   # ← v1
from langchain_chroma import Chroma                                    # ← v1
from langchain_huggingface import HuggingFaceEmbeddings                # ← v1
from langchain_ollama import ChatOllama
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.messages import HumanMessage, AIMessage
from langchain.chains import create_history_aware_retriever, create_retrieval_chain
from langchain.chains.combine_documents import create_stuff_documents_chain

# ── CONFIG PARETO v1 ──────────────────────────────────────────────────
OLLAMA_MODEL = “mistral”   # ollama pull mistral (une seule fois)
CHUNK_SIZE   = 500         # J4 : sweet spot
CHUNK_OVL    = 50          # J4 : 10% du chunk
TOP_K        = 3           # J3 : 3 chunks suffisent
MAX_HIST     = 5           # J6 : 5 tours max en mémoire

def build_chain(retriever):
“Construit la chaîne RAG conversationnelle (pattern LangChain v1)”
llm = ChatOllama(model=OLLAMA_MODEL, temperature=0)
ctx_p = ChatPromptTemplate.from_messages([
(“system”, “Reformule la question avec l’historique. Retourne uniquement la question.”),
MessagesPlaceholder(“chat_history”),
(“human”, ”{input}”),
])
qa_p = ChatPromptTemplate.from_messages([
(“system”, “Réponds UNIQUEMENT avec le contexte fourni. Si absent : ‘Information non disponible dans les documents.’
Cite toujours (Source : fichier.pdf).

Contexte :
{context}”),
MessagesPlaceholder(“chat_history”),
(“human”, ”{input}”),
])
return create_retrieval_chain(
create_history_aware_retriever(llm, retriever, ctx_p),
create_stuff_documents_chain(llm, qa_p)
)

# ── INTERFACE ─────────────────────────────────────────────────────────
st.set_page_config(page_title=“📚 RAG Entreprise”, layout=“wide”)
st.title(“📚 Chatbot Documentaire — Mistral + Ollama (LangChain v1)”)

with st.sidebar:
st.header(“📄 Documents”)
pdfs = st.file_uploader(“Uploader vos PDFs”, type=“pdf”, accept_multiple_files=True)
st.caption(“🟢 Mistral tourne 100% en local”)
if st.button(“🗑️ Nouvelle session”):
for k in [“chain”, “messages”, “history”]:
st.session_state.pop(k, None)
st.rerun()

# ── INDEXATION ────────────────────────────────────────────────────────
if pdfs and “chain” not in st.session_state:
with st.spinner(“Indexation en cours (embeddings locaux)…”):
all_chunks = []
splitter = RecursiveCharacterTextSplitter(chunk_size=CHUNK_SIZE, chunk_overlap=CHUNK_OVL)
for pdf in pdfs:
with tempfile.NamedTemporaryFile(delete=False, suffix=”.pdf”) as f:
f.write(pdf.read()); tmp = f.name
chunks = splitter.split_documents(PyPDFLoader(tmp).load())
for c in chunks: c.metadata[“source”] = pdf.name
all_chunks.extend(chunks)
vectordb = Chroma.from_documents(
all_chunks,
HuggingFaceEmbeddings(model_name=“all-MiniLM-L6-v2”)
)
st.session_state.chain    = build_chain(vectordb.as_retriever(search_kwargs={“k”: TOP_K}))
st.session_state.messages = []   # affichage Streamlit
st.session_state.history  = []   # HumanMessage/AIMessage — pattern v1
st.success(f”✓ {len(all_chunks)} chunks indexés depuis {len(pdfs)} PDF(s)”)

# ── CHAT ─────────────────────────────────────────────────────────────
for msg in st.session_state.get(“messages”, []):
with st.chat_message(msg[“role”]): st.markdown(msg[“content”])

if q := st.chat_input(“Posez votre question…”):
if “chain” not in st.session_state:
st.warning(“⬅️ Uploadez d’abord des PDFs”)
else:
st.session_state.messages.append({“role”: “user”, “content”: q})
with st.chat_message(“user”): st.markdown(q)
with st.chat_message(“assistant”), st.spinner(“Mistral réfléchit…”):
res = st.session_state.chain.invoke({
“input”: q,
“chat_history”: st.session_state.history
})
ans = res[“answer”]
st.markdown(ans)
sources = list({d.metadata.get(“source”, ”?”) for d in res.get(“context”, [])})
if sources: st.caption(f”📄 Sources : {’, ’.join(sources)}”)
st.session_state.messages.append({“role”: “assistant”, “content”: ans})
st.session_state.history.extend([HumanMessage(content=q), AIMessage(content=ans)])
st.session_state.history = st.session_state.history[-MAX_HIST * 2:]
```

Déploiement : Ollama en local vs cloud

Pour déployer sur Streamlit Cloud, Ollama ne sera pas disponible : tu devras soit utiliser une VM avec Ollama installé, soit switcher sur Groq/OpenAI uniquement pour la prod cloud. En local ou sur un serveur interne d'entreprise, Ollama est parfait.
