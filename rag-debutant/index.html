```
# Jour 1 — Tes premiers embeddings
```

from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

# Charger un modèle léger (gratuit, tourne en local)
model = SentenceTransformer(‘all-MiniLM-L6-v2’)

phrases = [
“Le chat dort sur le canapé”,
“Un félin se repose sur le sofa”,
“Python est un langage de programmation”,
“Il pleut beaucoup en automne”,
]

# Générer les embeddings
vecteurs = model.encode(phrases)
print(f”Forme du vecteur : {vecteurs.shape}”)  # (4, 384)

# Calculer la similarité entre la phrase 0 et toutes les autres
scores = cosine_similarity([vecteurs[0]], vecteurs)[0]
for i, score in enumerate(scores):
print(f”Similarité avec ‘{phrases[i][:30]}…’ : {score:.3f}”)

import chromadb
from sentence_transformers import SentenceTransformer

client = chromadb.PersistentClient(path=”./ma_base”)
collection = client.get_or_create_collection(“mes_docs”)
model = SentenceTransformer(‘all-MiniLM-L6-v2’)

documents = [
“L’IA générative révolutionne le développement logiciel”,
“Python est idéal pour le machine learning”,
“Docker facilite le déploiement des applications”,
“Le RAG améliore la précision des LLM sur des données privées”,
]

# Ajouter les documents avec leurs embeddings
embeddings = model.encode(documents).tolist()
collection.add(
documents=documents,
embeddings=embeddings,
ids=[f”doc_{i}” for i in range(len(documents))]
)

# Rechercher
query = “Comment améliorer la précision d’un chatbot ?”
results = collection.query(
query_embeddings=model.encode([query]).tolist(),
n_results=2
)
print(results[‘documents’])
